# DeepSeek RAG Chatbot Setup Guide

This guide provides a comprehensive walkthrough for setting up a production-ready, context-aware chatbot using **Next.js (App Router)**, **PostgreSQL (`pgvector`)**, **Prisma**, and the **DeepSeek API** via the **Vercel AI SDK**. 

The system leverages a hybrid architecture supporting both **Linear Retrieval-Augmented Generation (RAG)** and dynamic **Tool Calling** (Function Calling) without requiring LangChain.

## 1. Prerequisites

Ensure you have the following installed and configured before starting:
- **Node.js**: v18.x or higher
- **Next.js**: v14+ or v15+ (App Router)
- **PostgreSQL**: Instance with `pgvector` extension capabilities enabled (e.g., Supabase, Neon, or local Docker instance)
- **API Keys**: DeepSeek API key and an embedding model provider key (e.g., OpenAI API key for `text-embedding-3-small` or Cohere)

Install the core dependencies inside your Next.js project:

```bash
npm install @prisma/client @ai-sdk/openai ai zod pdf-parse dotenv
npm install -D prisma @types/pdf-parse

```

## 2. Database & Connection Setup

Ensure your PostgreSQL database instance has the `pgvector` extension enabled. Connect to your database using your preferred client or management console and execute the following query:


## 3. Prisma Integration

Initialize Prisma if you haven't already:

```bash
npx prisma init

```

Update your `prisma/schema.prisma` file. Since Prisma does not natively abstract vector columns completely out of the box, we use the `Unsupported` attribute to hook into the database's native `vector` type.

*Note: OpenAI's `text-embedding-3-small` model outputs a vector dimension size of `1536`.*

```prisma
datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

generator client {
  provider = "prisma-client-js"
}

model File {
  id        String       @id @default(uuid())
  name      String
  createdAt DateTime     **default(now())
  chunks    FileChunk[]
}

model FileChunk {
  id         String                      @id @default(uuid())
  fileId     String
  file       File                        @relation(fields: [fileId], references: [id], onDelete: Cascade)
  content    String                      // The textual segment snippet
  embedding  Unsupported("vector(1536)")? // Explicitly match your embedding model dimension size

  @@index([fileId])
}

```

Push the schema directly to your database:

```bash
npx prisma db push

```

Create a reusable database singleton instance inside `src/lib/prisma.ts`:

```typescript
import { PrismaClient } from '@prisma/client';

const globalForPrisma = globalThis as unknown as { prisma: PrismaClient | undefined };

export const prisma = globalForPrisma.prisma ?? new PrismaClient();

if (process.env.NODE_ENV !== 'production') globalForPrisma.prisma = prisma;

```

## 4. File Ingestion Pipeline (Chunking & Embedding)

Create a client utility to handle text vectorization in `src/lib/embeddings.ts`:

```typescript
import { OpenAI } from "openai";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

/**
 * Converts a string text slice into a high-dimensional mathematical vector.
 */
export async function getEmbedding(text: string): Promise<number[]> {
  const cleanText = text.replace(/\\n/g, " ");
  const response = await openai.embeddings.create({
    model: "text-embedding-3-small",
    input: cleanText,
    encoding_format: "float",
  });
  
  return response.data[0].embedding;
}

```

Create a Server Action or API framework utility to extract, split, and save text embeddings in `src/app/actions/ingest.ts`:

```typescript
"use server";

import { prisma } from "@/lib/prisma";
import { getEmbedding } from "@/lib/embeddings";
import pdfParse from "pdf-parse";

/**
 * Simple, standard text splitter utility (Recursive character chunking strategy alternative).
 */
function chunkText(text: string, chunkSize = 1000, chunkOverlap = 200): string[] {
  const chunks: string[] = [];
  let i = 0;
  
  while (i < text.length) {
    chunks.push(text.substring(i, i + chunkSize));
    i += chunkSize - chunkOverlap;
  }
  return chunks;
}

export async function ingestFile(formData: FormData) {
  const file = formData.get("file") as File;
  if (!file) throw new Error("No file uploaded.");

  const bytes = await file.arrayBuffer();
  const buffer = Buffer.from(bytes);
  let rawText = "";

  // Handle PDF parsing or raw TXT streams
  if (file.name.endsWith(".pdf")) {
    const pdfData = await pdfParse(buffer);
    rawText = pdfData.text;
  } else {
    rawText = buffer.toString("utf-8");
  }

  // Create document entry in database
  const dbFile = await prisma.file.create({
    data: { name: file.name },
  });

  const textChunks = chunkText(rawText);

  // Sequentially process chunks to prevent hitting downstream API rate-limits
  for (const chunk of textChunks) {
    if (!chunk.trim()) continue;
    
    const vector = await getEmbedding(chunk);
    const vectorString = `[${vector.join(",")}]`;

    // Write chunk structure and binary vector directly via SQL execute blocks
    await prisma.$executeRawUnsafe(`
      INSERT INTO "FileChunk" ("id", "fileId", "content", "embedding")
      VALUES (gen_random_uuid(), ${dbFile.id}, ${chunk}, CAST(${vectorString} AS vector));
    `);
  }

  return { success: true, fileId: dbFile.id };
}

```

## 5. Vector Database Search

Create a backend contextual abstraction utility to perform cosine distance verification searches inside `src/lib/vector-search.ts`:

```typescript
import { prisma } from "@/lib/prisma";
import { getEmbedding } from "@/lib/embeddings";

export async function searchUploadedFiles(query: string, matchThreshold = 0.6, matchCount = 4): Promise<string> {
  const queryEmbedding = await getEmbedding(query);
  const vectorString = `[${queryEmbedding.join(",")}]`;

  // Perform Cosine Similarity Search using the pgvector <=> operator
  const chunks = await prisma.$queryRawUnsafe<Array<{ content: string; similarity: number }>>(`
    SELECT "content", 1 - ("embedding" <=> CAST('${vectorString}' AS vector)) AS "similarity"
    FROM "FileChunk"
    WHERE 1 - ("embedding" <=> CAST('${vectorString}' AS vector)) > ${matchThreshold}
    ORDER BY "similarity" DESC
    LIMIT ${matchCount};
  `);

  if (!chunks || chunks.length === 0) {
    return "No relevant documents or file context snippets found in the database.";
  }

  return chunks.map(chunk => chunk.content).join("\\n\\n--- Next Snippet ---\\n\\n");
}

```

## 6. Chat API Route with Native Tool Calling

DeepSeek APIs mirror the OpenAI endpoint schema spec perfectly. Using the Vercel AI SDK, we point the native configuration adapter to `https://api.deepseek.com` and specify `deepseek-chat`.

Create the Next.js API endpoint in `src/app/api/chat/route.ts`:

```typescript
import { createOpenAI } from "@ai-sdk/openai";
import { streamText, tool } from "ai";
import { searchUploadedFiles } from "@/lib/vector-search";
import { z } from "zod";

// Configure standard Vercel AI OpenAI wrapper to interface directly with DeepSeek endpoints
const deepseek = createOpenAI({
  apiKey: process.env.DEEPSEEK_API_KEY,
  baseURL: "[https://api.deepseek.com/v1](https://api.deepseek.com/v1)", 
});

export async function POST(req: Request) {
  const { messages } = await req.json();

  const result = streamText({
    model: deepseek("deepseek-chat"),
    messages,
    system: `You are an expert file analysis chatbot. Users will upload documents. 
    When asked about specific documents or files, you must call the 'searchUploadedFiles' tool to access context before responding. 
    Always prioritize factual accuracy based strictly on the retrieved document segments.`,
    tools: {
      searchUploadedFiles: tool({
        description: "Searches through database records containing uploaded documents for context match metrics.",
        parameters: z.object({
          query: z.string().description("The context query parameters optimized for database retrieval match keys."),
        }),
        execute: async ({ query }) => {
          const context = await searchUploadedFiles(query);
          return { context };
        },
      }),
    },
    // Execute calls and forward the stream payload once final calculations wrap
    maxSteps: 3, 
  });

  return result.toDataStreamResponse();
}

```

## 7. Frontend Chat UI Components

Build your core chat application viewport leveraging the lightweight `useChat` react hooks from the Vercel AI package.

Create the frontend interface component in `src/app/chat/page.tsx`:

```tsx
"use client";

import { useChat } from "ai/react";
import { useState } from "react";
import { ingestFile } from "@/app/actions/ingest";

export default function ChatbotInterface() {
  const { messages, input, handleInputChange, handleSubmit, isLoading } = useChat({
    api: "/api/chat",
  });
  
  const [uploading, setUploading] = useState(false);
  const [status, setStatus] = useState("");

  const handleFileUpload = async (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file) return;

    setUploading(true);
    setStatus("Ingesting, parsing, and generating vector embeddings...");
    
    try {
      const formData = new FormData();
      formData.append("file", file);
      
      await ingestFile(formData);
      setStatus(`Successfully ingested ${file.name}!`);
    } catch (err) {
      console.error(err);
      setStatus("Error processing upload pipeline.");
    } finally {
      setUploading(false);
    }
  };

  return (
    <div style={{ maxWidth: "800px", margin: "0 auto", padding: "24px", fontFamily: "sans-serif" }}>
      <h1>DeepSeek Document Chatbot</h1>
      
      {/* File Upload Control Row */}
      <div style={{ background: "#f3f4f6", padding: "16px", borderRadius: "8px", marginBottom: "24px" }}>
        <label style={{ fontWeight: "bold", display: "block", marginBottom: "8px" }}>Upload Context Files (.txt or .pdf)</label>
        <input type="file" accept=".txt,.pdf" onChange={handleFileUpload} disabled={uploading} />
        {status && <p style={{ fontSize: "14px", marginTop: "8px", color: "#4b5563" }}>{status}</p>}
      </div>

      {/* Chat Conversational Container */}
      <div style={{ border: "1px solid #e5e7eb", borderRadius: "8px", padding: "16px", height: "400px", overflowY: "auto", marginBottom: "16px", background: "#fafafa" }}>
        {messages.map((m) => (
          <div key={m.id} style={{ marginBottom: "12px", textAlign: m.role === "user" ? "right" : "left" }}>
            <span style={{ display: "inline-block", background: m.role === "user" ? "#0070f3" : "#e5e7eb", color: m.role === "user" ? "#fff" : "#000", padding: "8px 12px", borderRadius: "12px", maxWidth: "80%", wordBreak: "break-word" }}>
              {m.content}
            </span>
          </div>
        ))}
        {isLoading && <p style={{ color: "#9ca3af", fontStyle: "italic" }}>Assistant is responding...</p>}
      </div>

      {/* Chat Prompt Entry Field */}
      <form onSubmit={handleSubmit} style={{ display: "flex", gap: "8px" }}>
        <input value={input} onChange={handleInputChange} placeholder="Ask a question about your files..." style={{ flexGrow: 1, padding: "10px", borderRadius: "6px", border: "1px solid #ccc" }} />
        <button type="submit" style={{ background: "#0070f3", color: "#fff", border: "none", padding: "10px 16px", borderRadius: "6px", cursor: "pointer" }} disabled={isLoading}>
          Send
        </button>
      </form>
    </div>
  );
}

```
